    <script>
        const library = {
            zh: {
                subjects: ["月色", "清风", "晚霞", "万物", "星河", "余生"],
                actions: ["不及你", "皆为你", "因你而温柔", "都藏着对你的告白"],
                quotes: ["浮世三千，吾爱有三：日、月与卿。", "你是四月早天里的云烟。", "山河远阔，人间烟火，无一是你，无一不是你。"]
            },
            en: ["You are my today and all of my tomorrows.", "I love you more than words can say.", "In your smile, I see something more beautiful than the stars."],
            fr: ["Je t'aime pour toujours.", "Tu es ma plus belle histoire d'amour.", "La vie est belle avec toi."]
        };

        function generate() {
            const dice = Math.random();
            let result = "";
            if (dice < 0.4) { // 动态组合中文
                const s = library.zh.subjects[Math.floor(Math.random() * library.zh.subjects.length)];
                const a = library.zh.actions[Math.floor(Math.random() * library.zh.actions.length)];
                result = s + a;
            } else if (dice < 0.7) { // 随机英文
                result = library.en[Math.floor(Math.random() * library.en.library.en.length)];
            } else { // 随机法语或中文名句
                const combined = [...library.zh.quotes, ...library.fr];
                result = combined[Math.floor(Math.random() * combined.length)];
            }
            document.getElementById('text').innerText = result;
        }
        generate(); // 运行生成
    </script>
